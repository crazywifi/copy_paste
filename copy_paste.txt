nmap -p- --open -n -Pn -T4 -sV <IP> -oG - | awk '
/open/ {
    split($1, a, "/"); 
    if ($0 ~ /https/) 
        print "https://" $2 ":" a[1]; 
    else 
        print "http://" $2 ":" a[1];
}' | tee urls.txt
